connection, _ := amqp.Dial(amqpURI)

channel, _ := connection.Channel()

for i := 0; i < 10; i++ {
    channel.Publish(
        "example", //exchange
        "hello.world", //routingKey
        false,
        false,
        amqp.Publishing{
            Headers: amqp.Table{},
            ContentType: "text/plain",
            ContentEncoding: "UTF-8",
            Body: []byte("Hi"),
            DeliveryMode: amqp.Transient,
            Priority: 0,
        },
    )
}
